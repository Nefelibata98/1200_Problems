##### 线性筛法

```cpp
using namespace std;

const int N = 10000;

int minp[N]; // i 的最小质数约束为 minp[i]  
int primes[N], cnt; // 记录所有质数
bool st[N];

// 质数线性筛法
int get_primes(int n)
{
    for(int i = 2; i < n; ++i)
    {
        if(!st[i]) 
        {
            primes[cnt++] = i;
            minp[i] = i;
        }
        for(int j = 0; primes[j] * i <= n; ++j)
        {
            int t = primes[j] * i;
            st[t] = true;
            minp[t] = primes[j];
            if(i % primes[j] == 0) break; 
        }
    }
}

int main(void)
{
    get_primes(10000);
    
    for(int i = 0; i < 20; ++i) cout << primes[i] << endl;
    
    return 0;
}
```
